const { exec } = require('child_process');

const startRobotSimulator = () => {
  return new Promise((resolve, reject) => {
    exec('bash /path/to/start-robot-simulator.sh', (error, stdout, stderr) => {
      if (error) {
        reject(error);
      } else {
        resolve(stdout);
      }
    });
  });
};

#!/bin/bash

# Install sshpass if it's not already installed
if ! command -v sshpass &> /dev/null
then
    sudo apt-get update && sudo apt-get install -y sshpass
fi

# Connect to the remote server and start the Robot Simulator
sshpass -p 'your-password-here' ssh user@server 'cd /path/to/robot/simulator && ./startRobotSimulator &'
